  SELECT COUNT(*) INTO v_public_function_grants                                                                                                                                                                                          
  FROM information_schema.role_routine_grants                                                                                                                                                                                            
  WHERE specific_schema = 'public'                                                                                                                                                                                                       
    AND grantee = 'public';                                                                                                                                                                                                              
                                                                                                                                                                                                                                         
  RAISE NOTICE 'SUMMARY|tables=%|rls_enabled=%|rls_disabled=%|policies=%|public_role_policies=%|anon_role_policies=%|unconditional_policies=%|anon_table_grants=%|public_table_grants=%|anon_function_grants=%|public_function_grants=%',
    v_total_tables, v_rls_enabled, v_rls_disabled, v_policy_count,                                                                                                                                                                       
    v_public_role_policies, v_anon_role_policies, v_unconditional_policies,                                                                                                                                                              
    v_anon_table_grants, v_public_table_grants, v_anon_function_grants, v_public_function_grants;                                                                                                                                        
                                                                                                                                                                                                                                         
  FOR r IN                                                                                                                                                                                                                               
    WITH policy_counts AS (                                                                                                                                                                                                              
      SELECT tablename, COUNT(*) AS cnt                                                                                                                                                                                                  
      FROM pg_policies                                                                                                                                                                                                                   
      WHERE schemaname = 'public'                                                                                                                                                                                                        
      GROUP BY tablename                                                                                                                                                                                                                 
    ),                                                                                                                                                                                                                                   
    grants AS (                                                                                                                                                                                                                          
      SELECT table_name,                                                                                                                                                                                                                 
             BOOL_OR(grantee = 'anon') AS anon_grant,                                                                                                                                                                                    
             BOOL_OR(grantee = 'public') AS public_grant,                                                                                                                                                                                
             BOOL_OR(grantee = 'authenticated') AS authenticated_grant                                                                                                                                                                   
      FROM information_schema.table_privileges                                                                                                                                                                                           
      WHERE table_schema = 'public'                                                                                                                                                                                                      
      GROUP BY table_name                                                                                                                                                                                                                
    )                                                                                                                                                                                                                                    
    SELECT c.relname AS table_name,                                                                                                                                                                                                      
           c.relrowsecurity AS rls_enabled,                                                                                                                                                                                              
           c.relforcerowsecurity AS force_rls,                                                                                                                                                                                           
           COALESCE(pc.cnt, 0) AS policy_count,                                                                                                                                                                                          
           COALESCE(g.anon_grant, FALSE) AS anon_grant,                                                                                                                                                                                  
           COALESCE(g.public_grant, FALSE) AS public_grant,                                                                                                                                                                              
           COALESCE(g.authenticated_grant, FALSE) AS authenticated_grant                                                                                                                                                                 
    FROM pg_class c                                                                                                                                                                                                                      
    JOIN pg_namespace n ON n.oid = c.relnamespace                                                                                                                                                                                        
    LEFT JOIN policy_counts pc ON pc.tablename = c.relname                                                                                                                                                                               
    LEFT JOIN grants g ON g.table_name = c.relname                                                                                                                                                                                       
    WHERE n.nspname = 'public' AND c.relkind = 'r'                                                                                                                                                                                       
    ORDER BY c.relname                                                                                                                                                                                                                   
  LOOP                                                                                                                                                                                                                                   
    RAISE NOTICE 'TABLE|%|rls=%|force=%|policies=%|anon_grant=%|public_grant=%|authenticated_grant=%',                                                                                                                                   
      r.table_name, r.rls_enabled, r.force_rls, r.policy_count,                                                                                                                                                                          
      r.anon_grant, r.public_grant, r.authenticated_grant;                                                                                                                                                                               
  END LOOP;                                                                                                                                                                                                                              
                                                                                                                                                                                                                                         
  FOR r IN                                                                                                                                                                                                                               
    SELECT tablename,                                                                                                                                                                                                                    
           policyname,                                                                                                                                                                                                                   
           cmd,                                                                                                                                                                                                                          
           roles::text AS roles_text,                                                                                                                                                                                                    
           LEFT(REPLACE(regexp_replace(COALESCE(qual, ''), '\s+', ' ', 'g'), '|', '/'), 280) AS qual_text,                                                                                                                               
           LEFT(REPLACE(regexp_replace(COALESCE(with_check, ''), '\s+', ' ', 'g'), '|', '/'), 280) AS check_text                                                                                                                         
    FROM pg_policies                                                                                                                                                                                                                     
    WHERE schemaname = 'public'                                                                                                                                                                                                          
    ORDER BY tablename, policyname                                                                                                                                                                                                       
  LOOP                                                                                                                                                                                                                                   
    RAISE NOTICE 'POLICY|%|%|cmd=%|roles=%|qual=%|check=%',                                                                                                                                                                              
      r.tablename, r.policyname, r.cmd, r.roles_text, r.qual_text, r.check_text;                                                                                                                                                         
  END LOOP;                                                                                                                                                                                                                              
                                                                                                                                                                                                                                         
  FOR r IN                                                                                                                                                                                                                               
    SELECT tablename,                                                                                                                                                                                                                    
           policyname,                                                                                                                                                                                                                   
           cmd,                                                                                                                                                                                                                          
           roles::text AS roles_text,                                                                                                                                                                                                    
           LEFT(REPLACE(regexp_replace(COALESCE(qual, ''), '\s+', ' ', 'g'), '|', '/'), 180) AS qual_text,                                                                                                                               
           LEFT(REPLACE(regexp_replace(COALESCE(with_check, ''), '\s+', ' ', 'g'), '|', '/'), 180) AS check_text                                                                                                                         
    FROM pg_policies                                                                                                                                                                                                                     
    WHERE schemaname = 'public'                                                                                                                                                                                                          
      AND (                                                                                                                                                                                                                              
        roles::text ILIKE '%public%'                                                                                                                                                                                                     
        OR roles::text ILIKE '%anon%'                                                                                                                                                                                                    
        OR lower(trim(COALESCE(qual, ''))) IN ('true', '(true)')                                                                                                                                                                         
        OR lower(trim(COALESCE(with_check, ''))) IN ('true', '(true)')                                                                                                                                                                   
      )                                                                                                                                                                                                                                  
    ORDER BY tablename, policyname                                                                                                                                                                                                       
  LOOP                                                                                                                                                                                                                                   
    RAISE NOTICE 'RISK_POLICY|%|%|cmd=%|roles=%|qual=%|check=%',                                                                                                                                                                         
      r.tablename, r.policyname, r.cmd, r.roles_text, r.qual_text, r.check_text;                                                                                                                                                         
  END LOOP;                                                                                                                                                                                                                              
                                                                                                                                                                                                                                         
  FOR r IN                                                                                                                                                                                                                               
    SELECT routine_name,                                                                                                                                                                                                                 
           grantee,                                                                                                                                                                                                                      
           privilege_type                                                                                                                                                                                                                
    FROM information_schema.role_routine_grants                                                                                                                                                                                          
    WHERE specific_schema = 'public'                                                                                                                                                                                                     
      AND grantee IN ('anon', 'public', 'authenticated')                                                                                                                                                                                 
    ORDER BY routine_name, grantee                                                                                                                                                                                                       
  LOOP                                                                                                                                                                                                                                   
    RAISE NOTICE 'ROUTINE_GRANT|%|grantee=%|priv=%', r.routine_name, r.grantee, r.privilege_type;                                                                                                                                        
  END LOOP;                                                                                                                                                                                                                              
                                                                                                                                                                                                                                         
  FOR r IN                                                                                                                                                                                                                               
    SELECT p.proname,                                                                                                                                                                                                                    
           pg_get_function_identity_arguments(p.oid) AS args,                                                                                                                                                                            
           p.prosecdef AS is_security_definer,                                                                                                                                                                                           
           has_function_privilege('anon', p.oid, 'EXECUTE') AS anon_exec,                                                                                                                                                                
           has_function_privilege('public', p.oid, 'EXECUTE') AS public_exec,                                                                                                                                                            
           has_function_privilege('authenticated', p.oid, 'EXECUTE') AS auth_exec                                                                                                                                                        
    FROM pg_proc p                                                                                                                                                                                                                       
    JOIN pg_namespace n ON n.oid = p.pronamespace                                                                                                                                                                                        
    WHERE n.nspname = 'public'                                                                                                                                                                                                           
    ORDER BY p.proname                                                                                                                                                                                                                   
  LOOP                                                                                                                                                                                                                                   
    RAISE NOTICE 'FUNCTION|%(% )|definer=%|anon_exec=%|public_exec=%|auth_exec=%',                                                                                                                                                       
      r.proname, r.args, r.is_security_definer, r.anon_exec, r.public_exec, r.auth_exec;                                                                                                                                                 
  END LOOP;                                                                                                                                                                                                                              
                                                                                                                                                                                                                                         
  RAISE EXCEPTION 'RLS_AUDIT_COMPLETE';                                                                                                                                                                                                  
END $$
